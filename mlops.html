		#@ZIP Class
		class getZip(getData):
			def ingest(self, file_path: str) -> pd.DataFrame:
			
				"""Extract .zip file & return pandas DataFrame.""" 
				#@check if file is a .zip 
				if not file_path.endswith(".zip"):
					raise ValueError ("File not a .zip.")
		
				#@If file is .zip extract
				with zipfile.ZipFile(file_path, "r") as zip:
					zip.extractall("files")
		
				# Find the extracted CSV file (assuming there is one CSV file inside the zip)
				files = os.listdir("files")
				csv_files = [file for files in files if file.endswith(".csv")]
		
				if len(csv_files) == 0:
					raise FileNotFoundError("No CSV file found.")
				if len(csv_files) > 1:
					raise ValueError("Multiple CSV files found. Please specify which one to use.")
		
				 # Read the CSV into a DataFrame
				csv_file_path = os.path.join("files", csv_files[0])
				df = pd.read_csv(csv_file_path)
		
				# Return the DataFrame
				return df
